\begin{setning}[Schwarz--Pick]
Látum $f \in \mathcal{O}(\mathbb{D})$ og gerum ráð fyrir að $f(\mathbb{D}) \subset \overline{\mathbb{D}}$.
Þá gildir
\[
\left|
  \frac{f(z_1)-f(z_2)}{1-\overline{f(z_1)}\,f(z_2)}
\right|
\le
\left|
  \frac{z_1-z_2}{1-\overline{z_1}\,z_2}
\right|,
\qquad z_1,z_2 \in \mathbb{D},
\]
og
\[
|f'(z)|
\le
\frac{1-|f(z)|^2}{1-|z|^2},
\qquad z \in \mathbb{D}.
\]
Jafnaðarmerki gildir í báðum ójöfnum ef $f$ er sjálfsmótun á $\mathbb{D}$.
\end{setning}
